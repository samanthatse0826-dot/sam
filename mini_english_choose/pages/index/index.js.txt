Page({
  data:{
    showModal:false,
    isRight:false,
    rightAns:"swing" //标准答案
  },
  //点击选项判题
  checkAnswer(e){
    let userAns = e.currentTarget.dataset.ans;
    let right = userAns === this.data.rightAns;
    this.setData({
      showModal:true,
      isRight:right
    })
  },
  //关闭弹窗
  closeModal(){
    this.setData({
      showModal:false
    })
  }
})
